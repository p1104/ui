ui

loading
    $("#loading").click(function () {
        up.ui.loading();
        setTimeout(function () {
            up.ui.hideLoading();
        },3000);
    })
toast
    $("#toast").click(function () {
        up.ui.toast("toast");
    })
alert
    $("#alert").click(function () {
        up.ui.alert({
             title:"alert title",
             msg:"alert msg",
             btns:[{
                 msg:"cancel",
                 callback:function(data){
                     $("#alert").text("you clicked cancel");
                 }
             },{
                 msg:"ok",
                 callback:function(data){
                     $("#alert").text("you clicked ok");
                 }
            }]
        });
    })
    
    备注：详细的接口文档参考：unionpay(建议pc或者pad阅读)
    1、ui模块只对外暴露方法，不提供样式的结构。
    2、实际使用中需要定制化的，可以在只有页面的样式文件中进行覆盖。
